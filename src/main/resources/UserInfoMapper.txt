namespace com.demo.mapper

import entity com.demo.data.entity.UserInfo

import java.util.List

search selectById(userInfo: UserInfo): UserInfo {
     select * from user_info where id = :userInfo.info.id
}

search selectByNames(nameList: mul String): mul UserInfo {
     select * from user_info where name in :nameList
}
